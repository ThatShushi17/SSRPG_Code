// Know yourself, know your enemy

var loopCount = 0

var prevFoe = foe
var prevFoeCount = foe.count
var prevFoeDistance = foe.distance

var lastFoeDeathTime = totaltime

var right
var left

var potId

var isPermaBuilding = false
var canBash = true
var canDash
canDash = item.GetCooldown("dash") <= 0 | item.GetCooldown("bash") <= 0 & canBash

var elementalShield = "ice A shield"
var elementalCrossbow = "ice D crossbow"
var elementalSwords = ["Frostmourne", "ice D sword"]

var permaCrossbow = "crossbow"

func AAC()
  ?item.left.state = 3
    equipL sword 1* +0
  ?item.right.state = 3
    ?totaltime % 2
      equipR bash
    :
      equipR dash

var moonCounter = 0
var moonWalkReload = []
func Moondial(pair)
  // >`0,1,@pair[0]@

  ?moonCounter % 3 = 0
    right = pair[0]
    left = moon
    moonWalkReload.Add(pair[0])
  ?moonCounter % 3 = 1
    right = pair[1]
    left = moon
    moonWalkReload.Insert(0, pair[1])
  ?moonCounter % 3 = 2
    right = pair[0]
    left = pair[1]
    moonWalkReload.Clear()

  moonCounter++

var incomingSpit1 = 0
var incomingSpit2 = 0
func CheatSnake()
  ?foe.state = 31 | foe.state = 33 & foe.time = 0
    ?incomingSpit1 > totaltime
      incomingSpit2 = totaltime + 25
    :
      incomingSpit1 = totaltime + 25

  right = mask

  ?foe.distance <= 10 & item.GetCooldown(mind) <= 0 & (incomingSpit1 = totaltime + 25 | incomingSpit2 = totaltime + 25)
    left = mind

  :?incomingSpit1 > totaltime
    ?incomingSpit1 - totaltime >= 6
      left = permaCrossbow
    :
      left = trisk
      ?foe.distance <= 10 & foe.distance > 6
        right = "sword 1* +0"

  :?foe.distance <= 22
    left = permaCrossbow

  :
    left = quest

  ?incomingSpit1 = totaltime & incomingSpit2 > incomingSpit1
    incomingSpit1 = incomingSpit2

func Begin()
  right = comp
  left = trisk

  brew bronze + stone
  potId = "lucky_crit"

func Loop()
  right = comp
  left = trisk

  isPermaBuilding = false

func Cutscene()
  ?isPermaBuilding
    right = mask
  :
    right = comp
  
  ?ai.walking
    left = trisk
  :
    left = quest

var walkStutterstepItem = elementalCrossbow
func Walk()
  ?foe.distance > 10
    left = trisk

    ?foe.distance <= 16 & item.GetCooldown("bash") <= 0 & canBash
      right = bash
    :?foe.distance <= 16 & item.GetCooldown("dash") <= 0
      right = dash
    :
      ?item.GetCooldown("quarterstaff") & item.CanActivate()
        instantActives.Add("quarterstaff")
      
      ?pickup.distance < 18
        right = star
      
      :?foe.distance <= 40
        right = elementalShield
      
      :?moonWalkReload.Count() >= 1
        right = moonWalkReload[0]
        moonWalkReload.RemoveAt(0)
      
      :?isPermaBuilding
        right = mask
      
      :
        right = comp

    return true
  :
    ?totaltime % 2 = 0
      left = trisk
      right = comp
    
    :
      left = ""
      right = walkStutterstepItem


func Boss()
  ?buffs.GetTime(potId) <= 100
    isPermaBuilding = true
    activate P

  ?isPermaBuilding
    CheatSnake()
    return

  ?foe.distance > 6
    Walk()
    return

  ?item.potion ! empty
    activate P

  Moondial(elementalSwords)

func Miniboss()
  ?foe.distance > 6
    canBash = true
    Walk()
    return

  ?foe.debuffs.GetTime("stun") <= 5
    right = moon
    left = grappl
    return

  Moondial(elementalSwords)

var movedToMobs = false
func Mobs()
  ?foe.distance > 6 & !movedToMobs
    canBash = false
    ?screen.i > 1  // FIXME
      canBash = true

    Walk()
    return
  
  ?item.right.state = 3 & (foe.distance > 20 | canDash) |
  ^math.Abs(foe.z - pos.z) > 1 & totaltime - lastFoeDeathTime <= 1 |
  ^foe.count = 0
    movedToMobs = false
  :
    movedToMobs = true
  
  left = ""
  right = elementalCrossbow

func Conditional()
  ?(buffs.GetTime("infernal") - 1) % 60 = 0
    right = towering
    return true  // TODO make this less intrusive?

// \ ========-
//  \   Main
//   \ ========-
// The culmination of your run

func Main()
  ?loc.begin
    Begin()

  ?loc.loop
    loopCount++
    Loop()
    return

  // TODO add a Priority callback for hhammer and like

  ?Conditional()
    return

  ?ai.paused | !ai.enabled
    Cutscene()
    return

  ?foe = phase
    Boss()
    return

  ?foe = boss
    Miniboss()
    return

  ?encounter.isElite
    return

  Mobs()


// \ ==========-
//  \   Engine
//   \ ==========-
// You're naught but a cog, young one
// Dance like one

func Engine()
  right = ""
  left = ""

  Main()
  InstantAbilities()
  AAC()
  FinalEquips()
  FinalAbilities()
  UpdatePrevStats()

var instantActives = []
func InstantAbilities()
  var count = instantActives.Count()
  ?count = 0
    return
  
  for i = 0 .. count - 1
    var item = instantActives[i]
    ?Type(item) = "string"
      equip @item@
      activate R
    :
      var l = item[0]
      equipR @item[1]@
      activate R
      ?l
        equipL @l@
        activate L
  
  instantActives.Clear()

func FinalEquips()
  ?!left
    equip @right@
  :
    equipR @right@
    equipL @left@

var activateR = false
var activateL = false
func FinalAbilities()
  ?activateR
    activate R
    activateR = false
  
  ?activateL
    activate L
    activateL = false


func UpdatePrevStats()
  prevFoe = foe
  prevFoeCount = foe.count
  prevFoeDistance = foe.distance

  ?foe.state = 3
    lastFoeDeathTime = totaltime


Engine()
